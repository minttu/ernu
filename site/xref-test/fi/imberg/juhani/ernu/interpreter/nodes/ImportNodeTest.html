12  public class ImportNodeTest {
13      @Rule
14      public ExpectedException thrown = ExpectedException.none();
15      Environment environment;

17      @Before
18      public void init() {
19          environment = new Environment(false, null, "test");

22      @After
23      public void destroy() {
24          environment = null;
25      }
26  
27      public void hasSymbol(String string, Class cls) throws RuntimeException {
28          Node node = environment.getSymbol(string);
29          Assert.assertTrue(node.getClass().isAssignableFrom(cls));
30      }
31  
32      @Test
33      public void testImportSimple() throws RuntimeException {
34          ImportNode importNode = new ImportNode("math", "", new ArrayList<>());
35          importNode.getValue(environment);
36          hasSymbol("math", EnvironmentNode.class);
37      }
38  
39      @Test
40      public void testImportSimpleAs() throws RuntimeException {
41          ImportNode importNode = new ImportNode("math", "m", new ArrayList<>());
42          importNode.getValue(environment);
43          hasSymbol("m", EnvironmentNode.class);
44      }
45  
46      @Test
47      public void testImportSub() throws RuntimeException {
48          List<String> subs = new ArrayList<>();
49          subs.add("sqrt");
50          subs.add("pow");
51          ImportNode importNode = new ImportNode("math", "", subs);
52          importNode.getValue(environment);
53          hasSymbol("sqrt", ProxyNode.class);
54          hasSymbol("pow", ProxyNode.class);
55      }
56  
57      @Test
58      public void throwsNoSuchFile() throws RuntimeException {
59          thrown.expect(RuntimeException.class);
60          thrown.expectMessage("No such file as tfsne");
61  
62          ImportNode importNode = new ImportNode("tfsne", "", new ArrayList<>());
63          importNode.getValue(environment);
