20  public class PrintFunctionTest {
21      Environment environment;
22      PrintFunction printFunction;
23      ByteArrayOutputStream out;
24      PrintStream originalOut;
25  
26      @Before
27      public void init() {
28          environment = new Environment(false, null, "test");
29          printFunction = new PrintFunction();
30          out = new ByteArrayOutputStream();
31          originalOut = System.out;
32          System.setOut(new PrintStream(out));
33      }
34  
35      @After
36      public void destroy() {
37          environment = null;
38          printFunction = null;
39          System.setOut(originalOut);
40      }
41  
42      public void doTest(Node[] args, String expected) throws RuntimeException {
43          List<Node> argsList = new ArrayList<>();
44          Collections.addAll(argsList, args);
45          Node node = printFunction.call(environment, argsList);
46          Assert.assertTrue(node instanceof NullNode);
47          Assert.assertEquals(expected + "\n", out.toString());
48          out.reset();
49      }
50  
51      @Test
52      public void simpleTest() throws RuntimeException {
53          doTest(new Node[]{new NumberNode(0)}, "0");
54          doTest(new Node[]{new NumberNode(0), new NumberNode(1)}, "0, 1");
55          doTest(new Node[]{new StringNode("asd"), new NullNode()}, "asd, null");
56          doTest(new Node[]{}, "");
57      }
