17      public ExpectedException thrown = ExpectedException.none();
18      Environment environment;

21      @Before
22      public void init() {
23          environment = new Environment(false, null, "test");

28      public void destroy() {

33      @Test
34      public void takesExactlyOneArgument() throws RuntimeException {

38          lenFunction.call(environment, new ArrayList<>());
39      }
40  
41      @Test
42      public void requiresAObject() throws RuntimeException {
43          thrown.expect(RuntimeException.class);
44          thrown.expectMessage("lens argument should be a sequence");
45  
46          List<Node> nodeList = new ArrayList<>();
47          nodeList.add(new NullNode());
48  
49          lenFunction.call(environment, nodeList);
50      }
51  
52      @Test
53      public void works() throws RuntimeException {
54          List<Node> nodeList = new ArrayList<>();
55          nodeList.add(new StringNode("123"));
56          Node node = lenFunction.call(environment, nodeList);
57          Assert.assertTrue(node instanceof NumberNode);
58          Assert.assertEquals(3, ((NumberNode) node).getInteger());
