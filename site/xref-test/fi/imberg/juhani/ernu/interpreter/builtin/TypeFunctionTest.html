37          typeFunction.call(environment, new ArrayList<>());
38      }
39  
40      @Test
41      public void works() throws RuntimeException {
42          List<Node> args = new ArrayList<>();
43          args.add(new NullNode());
44          Node node = typeFunction.call(environment, args);

46          Assert.assertEquals("NullNode", ((StringNode) node).getStringLiteral());
