3   import fi.imberg.juhani.ernu.cli.exceptions.FileNotFoundException;

5   import java.io.IOException;
6   import java.nio.charset.StandardCharsets;
7   import java.nio.file.Files;
8   import java.nio.file.Paths;
9   
10  public abstract class FileCommand implements Command {
11      /**
12       * Retuns a string representation of the given filename
13       *
14       * @param fileName The file to open
15       * @return The contents of the file
16       * @throws FileNotFoundException If no such file is found
17       */
18      public String fileToString(String fileName) throws FileNotFoundException {
19          byte[] bytes = new byte[0];
20          try {
21              bytes = Files.readAllBytes(Paths.get(fileName));
22          } catch (IOException ex) {
23              throw new FileNotFoundException(fileName);
24          }
25          return new String(bytes, StandardCharsets.UTF_8);
26      }
27  }
